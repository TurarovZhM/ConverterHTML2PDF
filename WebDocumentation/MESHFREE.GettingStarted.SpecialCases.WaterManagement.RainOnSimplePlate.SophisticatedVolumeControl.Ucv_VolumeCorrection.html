Ucv_VolumeCorrection

implementation of the volume correction as a Ucv-procedure

#------------------------------------------------------------------------- # call: # include_Ucv{Ucv_VolumeCorrection.dat} parameters{ _WEIGHT_FrS_=1.0, ... # _WEIGHT_ISO_=0.5, ... # _WEIGHT_INT_=0.0, ... # _CHAMBER_=1, ... # _MATERIAL_=Mat1, ... # _CORRECTION_=0.001, ... # _Vdot_=constant or equation } # #------------------------------------------------------------------ # parameters: # _WEIGHT_FrS_ -> weight for volume correction by "lifting" free surface points along their surface normal # _WEIGHT_ISO_ -> weight for volume correction by adapting the volume partitions of isolated points # _WEIGHT_INT_ -> weight for volume correction by providing an adapted divergence of velocity; # ATTENTION: this does not work yet as the divergenceV(...) functionality does not really apply for this purpose # _CHAMBER_ -> provide chamber index # _MATERIAL_ -> provide the name of the material (DO NOT PUT IT INSIDE $...$) # _CORRECTION_ -> correction volume per time step shall not exceed this partition of the total volume # _Vdot_ -> positive number, signifies the upper bound of volume per time unit to be deleted by EVENT or METAPLANES, # the correction is based now on a target volume, that takes into account this limitation # it should be most probably be the same as the volume flux used for the droplet source #------------------------------------------------------------------------- # algorithm: # 1. produce current volume V=sum(V_i) # 2. produce potential volume for each FPM point V_pot_i # 3. produce defect volume to be corrected within this time step V_defect = min( _CORRECTION_*V , max( -_CORRECTION_*V , Vtarget-V ) ) # 4. produce factor alpha such that the defect volume is represented as V_defect = alpha*sum( V_pot_i ) # 5. move free surface such that the defect volume partition is produced alpha*V_pot_i = d_move*A_i*(0.3*h_i) # 6. adapt the V_i of isolated points by V_i = V_i + alpha*V_pot_i # (7.) provide a divergence of velocity such that in the next time cycle, volume is corrected by expansion/depression #------------------------------------------------------------------------- #--------------------------------------------------------------------------- # restrict the mass deleted at EVENT or METAPLANE by the given formulation for Vdot # has to be first, otherwise, in the first time cycle, dont know what is going to happen #--------------------------------------------------------------------------- include_Ucv{../includeFiles/Ucv_ComputeAdaptedTargetVolume.dat} parameters{ _Vdot_=_Vdot_, ... _CHAMBER_=_CHAMBER_, ... _Ncheck_=100, ... _ATV_AdaptedTarget_=$VtargetAdapted$ } #--------------------------------------------------------------------------- # algorithm implemented #--------------------------------------------------------------------------- begin_timestepfile{"VOCO_chamber=_CHAMBER__material=_MATERIAL_"} INTEGRATION($VOCO_time__MATERIAL_$) = ( %PUBLICVALUE%, [real(%RealTimeSimulation%)], %INTEGRATION_Header%, "time" ) INTEGRATION($VOCO_trueVolume__MATERIAL_$) = ( %INTEGRATION_INT%, 1, $_MATERIAL_$, %INTEGRATION_Header%, "VOCO_trueVolume__MATERIAL_" ) INTEGRATION($VOCO_potentialVolume__MATERIAL_$) = ( %INTEGRATION_INT%, [equn($EQN_potentialVolume__MATERIAL_$)/Y%ind_Vi%], $_MATERIAL_$, %INTEGRATION_Header%, "VOCO_potentialVolume__MATERIAL_" ) INTEGRATION($VOCO_defectVolume__MATERIAL_$) = ( %PUBLICVALUE%, [equn($EQN_defectVolume__MATERIAL_$)], %INTEGRATION_Header%, "VOCO_defectVolume__MATERIAL_" ) INTEGRATION($VOCO_alphaCorrection__MATERIAL_$) = ( %PUBLICVALUE%, [integ($VOCO_defectVolume__MATERIAL_$)/integ($VOCO_potentialVolume__MATERIAL_$)], %INTEGRATION_Header%, "VOCO_alphaCorrection__MATERIAL_" ) INTEGRATION($VOCO_checkFS__MATERIAL_$) = ( %ASSIGN_FUNCTIONVALUE%, %indU_checkFS%, [equn($EQN_chekFS__MATERIAL_$)], $_MATERIAL_$, %INTEGRATION_Header%, "VOCO_checkFS__MATERIAL_" ) # only quick visualization #INTEGRATION($VOCO_correctVi_INT__MATERIAL_$) = ( %ASSIGN_FUNCTIONVALUE%, %indU_corrVol%, [equn($EQN_correctionVolume_INT__MATERIAL_$)], $_MATERIAL_$, %INTEGRATION_Header%, "VOCO_correctVi_INT__MATERIAL_" ) INTEGRATION($VOCO_correct_x_FrS__MATERIAL_$) = ( %ASSIGN_FUNCTIONVALUE%, %ind_x(1)%, [Y%ind_x(1)%-equn($EQN_distMoveFrS__MATERIAL_$)*Y%ind_n(1)%], $_MATERIAL_$, %INTEGRATION_Header%, "VOCO_correct_x_FrS__MATERIAL_" ) INTEGRATION($VOCO_correct_y_FrS__MATERIAL_$) = ( %ASSIGN_FUNCTIONVALUE%, %ind_x(2)%, [Y%ind_x(2)%-equn($EQN_distMoveFrS__MATERIAL_$)*Y%ind_n(2)%], $_MATERIAL_$, %INTEGRATION_Header%, "VOCO_correct_y_FrS__MATERIAL_" ) INTEGRATION($VOCO_correct_z_FrS__MATERIAL_$) = ( %ASSIGN_FUNCTIONVALUE%, %ind_x(3)%, [Y%ind_x(3)%-equn($EQN_distMoveFrS__MATERIAL_$)*Y%ind_n(3)%], $_MATERIAL_$, %INTEGRATION_Header%, "VOCO_correct_z_FrS__MATERIAL_" ) INTEGRATION($VOCO_correctVi_ISO__MATERIAL_$) = ( %ASSIGN_FUNCTIONVALUE%, %ind_Vi%, [Y%ind_Vi%+equn($EQN_correctionVolume_ISO__MATERIAL_$)], $_MATERIAL_$, %INTEGRATION_Header%, "VOCO_correctVi_ISO__MATERIAL_" ) INTEGRATION($VOCO_adaptedTargetVolume__MATERIAL_$) = ( %PUBLICVALUE%, [integ($VtargetAdapted$)], %INTEGRATION_Header%, "VOCO_adaptedTargetVolume__MATERIAL_" ) end_timestepfile # define the correction divergence of velocity -> commented out since divergenceV() will, itself, adapt the target volume #divergenceV($_MATERIAL_$) = [ Y%indU_corrVol%/(Y%ind_dt%*Y%ind_Vi%) ] #SAVE_ITEM = ( %SAVE_scalar%, [ Y%indU_corrVol%/(Y%ind_dt%*Y%ind_Vi%) ], "correction_divergence" ) #-------------------------------------------------------------------- # helpful equations #-------------------------------------------------------------------- begin_equation{"EQN_weight_FrS__MATERIAL_"} # put given weight to regular free surface particles if ( (Y%ind_Organize%=%ORGANIZE_IsIsolated%) ) :: 0 else :: if ( Y%ind_kob%=%BND_free% ) :: _WEIGHT_FrS_ * equn($EQN_chekFS__MATERIAL_$) else :: 0 endif endif end_equation begin_equation{"EQN_weight_INT__MATERIAL_"} # put given weight to bulk particles if ( (Y%ind_Organize%=%ORGANIZE_IsIsolated%) ) :: 0 else :: _WEIGHT_INT_ endif end_equation begin_equation{"EQN_weight_ISO__MATERIAL_"} # put given weight to isolated particles if ( (Y%ind_Organize%=%ORGANIZE_IsIsolated%) ) :: _WEIGHT_ISO_ else :: 0 endif end_equation begin_equation{"EQN_potentialVolume__MATERIAL_"} # provide potential correction volume for each FPM point equn($EQN_weight_FrS__MATERIAL_$)*( (0.3*Y%ind_h%)*Y%ind_dA% ) + equn($EQN_weight_INT__MATERIAL_$)*( Y%ind_Vi% ) + equn($EQN_weight_ISO__MATERIAL_$)*( Y%ind_Vi% ) end_equation begin_equation{"EQN_defectVolume__MATERIAL_"} # defect volume to be corrected in this time step: defect volume, but restricted to min( _CORRECTION_*integ($VOCO_trueVolume__MATERIAL_$) , max( -_CORRECTION_*integ($VOCO_trueVolume__MATERIAL_$) , # (real(%FPM_VOLUME_TARGET%,_CHAMBER_)-real(%FPM_VOLUME_ACTUAL%,_CHAMBER_)) (integ($VtargetAdapted$)-real(%FPM_VOLUME_ACTUAL%,_CHAMBER_)) ) ) end_equation begin_equation{"EQN_correctionVolume_INT__MATERIAL_"} # correction volume for regular (interior) points integ($VOCO_alphaCorrection__MATERIAL_$)*equn($EQN_weight_INT__MATERIAL_$) * ( Y%ind_Vi% ) end_equation begin_equation{"EQN_distMoveFrS__MATERIAL_"} # correction distance for regular free surface points integ($VOCO_alphaCorrection__MATERIAL_$)*equn($EQN_weight_FrS__MATERIAL_$) * ( 0.3*Y%ind_h% ) end_equation begin_equation{"EQN_correctionVolume_ISO__MATERIAL_"} # correction volume for isolated points max(-0.5,min(0.5, integ($VOCO_alphaCorrection__MATERIAL_$)*equn($EQN_weight_ISO__MATERIAL_$) ))* ( Y%ind_Vi% ) #integ($VOCO_alphaCorrection__MATERIAL_$)*equn($EQN_weight_ISO__MATERIAL_$) * ( Y%ind_Vi% ) end_equation begin_equation{"EQN_chekFS__MATERIAL_"} # suggestion: this function emphasizes those (free surface) particles which are in a thin layer along a rigid wall max( -(Y%ind_sha(1)%*Y%ind_n(1)%+Y%ind_sha(2)%*Y%ind_n(2)%+Y%ind_sha(3)%*Y%ind_n(3)%)*(0.1*Y%ind_h%)/(Y%ind_dtb%+0.1*Y%ind_h%) , 0 ) + 0.5 end_equation
