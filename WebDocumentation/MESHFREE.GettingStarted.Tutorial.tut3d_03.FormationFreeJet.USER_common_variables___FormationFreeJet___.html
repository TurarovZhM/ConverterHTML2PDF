smoothing length of water (in m) end_alias #---------------------- # Solver Section #---------------------- begin_alias{} begin_selection{"TURB"} case{"OFF"} "TurbulenceModel" = " " case{"ON"} "TurbulenceModel" = "TURBULENCE:k-epsilon" end_selection end_alias KOP(1) = LIQUID LAGRANGE V:IMPLICIT vp- T:NONE &TurbulenceModel& #--------------------------------- # inflow in normal direction INITDATA($Mat1$,%ind_v(3)%) = [ &v0&*(Y%ind_n(3)%) ] # inflow in normal direction INITDATA($Mat1$,%ind_p%) = 0.0 INITDATA($Mat1$,%ind_p_dyn%) = 0.0 INITDATA($Mat1$,%ind_k%) = 0.001 INITDATA($Mat1$,%ind_eps%) = 1.0 #---------------------------------- #
