USER_common_variables

simple box driving through a channel of water: USER_common_variables.dat

See especially the first lines where we establish several instances of the box-in-channel example by include_Ucv{} .
#------------------------------------------------------------------------------------ # - this example shows the differences between various types of dynamic boundary conditions for walls # - simple box in channel # - apply the following boundary conditions at the box: (see FPM.InputFiles.USER_common_variables.BoundaryConditions.BCON.%ind_p_dyn% ): # %BND_none% # %BND_wall% # %BND_slip% # %BND_NEUMANN% # %BND_AVERAGE% # - write out measured pressure values at the front face of the box (see the .timestep files produced by FPM) # - compare these values to the Bernoulli values #------------------------------------------------------------------------------------ include_Ucv{Ucv_SinglePoolWithBox.dat} parameters{_CH_=1, _BC_=BND_none} # load the box-in-channel geometry into chamber 1 and apply %BND_none% boundary conditions for %ind_p_dyn% include_Ucv{Ucv_SinglePoolWithBox.dat} parameters{_CH_=2, _BC_=BND_wall} # load the box-in-channel geometry into chamber 2 and apply %BND_wall% boundary conditions for %ind_p_dyn% include_Ucv{Ucv_SinglePoolWithBox.dat} parameters{_CH_=3, _BC_=BND_slip} # load the box-in-channel geometry into chamber 3 and apply %BND_slip% boundary conditions for %ind_p_dyn% include_Ucv{Ucv_SinglePoolWithBox.dat} parameters{_CH_=4, _BC_=BND_NEUMANN} # load the box-in-channel geometry into chamber 2 and apply %BND_NEUMANN% boundary conditions for %ind_p_dyn% include_Ucv{Ucv_SinglePoolWithBox.dat} parameters{_CH_=5, _BC_=BND_AVERAGE} # load the box-in-channel geometry into chamber 2 and apply %BND_AVERAGE% boundary conditions for %ind_p_dyn% #cccccccccccccccccccccccccccccccccccccccccccccccccccccccccc #c Alias Section #cccccccccccccccccccccccccccccccccccccccccccccccccccccccccc begin_alias{} # maximum smoothing length "InitialWaterLevelAboveGround" = "0.5" # z-position of the initial water level "InitialOffsetOfBox" = "1, 0, 0.20" # initial placement of box "SpeedOfBox" = "4.0" # movement velocity of the box "ScalingOfPool" = "1, 1, 1" #
