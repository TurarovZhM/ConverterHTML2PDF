Ucv_ComputeAdaptedTargetVolume

recompute the target volume due to a given maximum volume flux

# adapt the target volume (for volume correction) due to a restriction of the volume per time that can be deleted by EVENT or METAPLANES #-------------------------------------------------------------------------------- # abbreviations: # ATV = adapted target volume # VDELAEMP = volume deleted at events and metaplanes #----------------------------------------------------------------------- # call: # include_Ucv{Ucv_ComputeAdaptedTargetVolume.dat} parameters{ _Vdot_=_Vdot_, ... # _CHAMBER_=1, ... # _Ncheck_=100, ... # _ATV_AdaptedTarget_=$VtargetAdapted$ } #------------------------------------------------------------------------ # parameters: # _Vdot_ = equation or constant value describing the limiting volume flux allowed at EVENT and/or METAPLANE cut # _CHAMBER_ = chamber index to apply this function to # _Ncheck_ = benchmark the state of EVENT-cut after _Ncheck_ time cycles, we balance the EVENT- and METPLANE-cut over this period # _ATV_AdaptedTarget_ = integration index where to put the adapted target volume #----------------------------------------------------------------------- begin_timestepfile{"adaptedTargetVolume_chamber=_CHAMBER_"} INTEGRATION($ATV_CHAMBER__INT_Time_$) = ( %PUBLICVALUE%, [real(%RealTimeSimulation%)], %INTEGRATION_Header%, "time" ) INTEGRATION($ATV_CHAMBER__INT_AllowedVDELAEMP_LastBenchmark$) = ( %PUBLICVALUE%, [equn($ATV_CHAMBER__EQN_AllowedVDELAEMP_LastBenchmark$)], %INTEGRATION_Header%, "allowed volume deleted by EVENT LastBenchmark" ) INTEGRATION($ATV_CHAMBER__INT_Time_LastBenchmark$) = ( %PUBLICVALUE%, [equn($ATV_CHAMBER__EQN_Time_LastBenchmark$)], %INTEGRATION_Header%, "time LastBenchmark" ) INTEGRATION($ATV_CHAMBER__INT_VDELAEMP$) = ( %PUBLICVALUE_SUM%, [real(%FPM_VOLUME_DeletedAtMetaplanes%,_CHAMBER_)], %INTEGRATION_Header%, "measured volume deleted by EVENT" ) INTEGRATION($ATV_CHAMBER__INT_VDELAEMP_LastBenchmark$) = ( %PUBLICVALUE%, [equn($ATV_CHAMBER__EQN_VDELAEMP_LastBenchmark$)], %INTEGRATION_Header%, "measured volume deleted by EVENT LastBenchmark" ) INTEGRATION($ATV_CHAMBER__INT_AllowedVDELAEMP$) = ( %PUBLICVALUE%, [equn($ATV_CHAMBER__EQN_AllowedVDELAEMP$)], %INTEGRATION_Header%, "allowed volume deleted by EVENT" ) INTEGRATION(_ATV_AdaptedTarget_) = ( %PUBLICVALUE%, [equn($ATV_CHAMBER__EQN_VTargetAdapted$)], %INTEGRATION_Header%, "adapted target volume" ) end_timestepfile # simulation time at the last benchmark begin_equation{$ATV_CHAMBER__EQN_Time_LastBenchmark$} if (mod(real(%NumberTimeStepsExecuted%),_Ncheck_)=0) :: real(%RealTimeSimulation%) else :: integ($ATV_CHAMBER__INT_Time_LastBenchmark$) endif end_equation # truely measured outflow volume at the last benchmark begin_equation{$ATV_CHAMBER__EQN_VDELAEMP_LastBenchmark$} if (mod(real(%NumberTimeStepsExecuted%),_Ncheck_)=0) :: integ($ATV_CHAMBER__INT_VDELAEMP$) else :: integ($ATV_CHAMBER__INT_VDELAEMP_LastBenchmark$) endif end_equation # allowed outflow volume at the last benchmark begin_equation{$ATV_CHAMBER__EQN_AllowedVDELAEMP_LastBenchmark$} if (mod(real(%NumberTimeStepsExecuted%),_Ncheck_)=0) :: integ($ATV_CHAMBER__INT_AllowedVDELAEMP$) else :: integ($ATV_CHAMBER__INT_AllowedVDELAEMP_LastBenchmark$) endif end_equation # the allowed outflow volume (deleted by EVENT or METAPLANE) is the minimum of the true outflow and the Vdot since last benchmark begin_equation{$ATV_CHAMBER__EQN_AllowedVDELAEMP$} integ($ATV_CHAMBER__INT_AllowedVDELAEMP_LastBenchmark$) + min( _Vdot_*(real(%RealTimeSimulation%) - integ($ATV_CHAMBER__INT_Time_LastBenchmark$)) , integ($ATV_CHAMBER__INT_VDELAEMP$)-integ($ATV_CHAMBER__INT_VDELAEMP_LastBenchmark$) ) end_equation # from the target volume, subtract the truly measured outflow volume, and add the allowed outflow begin_equation{$ATV_CHAMBER__EQN_VTargetAdapted$} real(%FPM_VOLUME_TARGET%,_CHAMBER_) + integ($ATV_CHAMBER__INT_VDELAEMP$) - integ($ATV_CHAMBER__INT_AllowedVDELAEMP$) end_equation
